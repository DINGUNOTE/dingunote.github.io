function solution(score) {
  const answer = Array(score.length).fill(1);
  const arr = score.map((v) => Math.floor(v[0] + v[1]) / 2);

  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length; j++) {
      arr[j] > arr[i] ? answer[i]++ : 0;
    }
  }

  return answer;
}
